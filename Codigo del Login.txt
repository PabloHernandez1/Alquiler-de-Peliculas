        private void bunifuImageButton1_Click(object sender, EventArgs e)
        {
            string usuario = textBox1.Text;
            string password = textBox2.Text;

            if (usuario == "" && password == "")
            {
                MessageBox.Show("Falta usuario y password");
                return;
            }
            else if (usuario == "")
            {
                MessageBox.Show("Falta usuario");
                return;
            }
            else if (password == "")
            {
                MessageBox.Show("Falta contraseña");
                return;
            }

            con.Open();
   
            OleDbDataAdapter da = new OleDbDataAdapter("SELECT * FROM usuarios WHERE usuario ='" + textBox1.Text + "'and password='" + textBox2.Text + "'", con);
            DataTable dt = new DataTable();
            da.Fill(dt);
            int contador = dt.Rows.Count;
            if (contador > 0)
            {

                string tipo = dt.Rows[0]["tipo_usuario"].ToString();
                if (tipo == "1")
                {
                    this.Hide();
                    Menu frm = new Menu();
                    frm.Show();
                }
                else if (tipo == "2")
                {
                    this.Hide();
                    Menu frm1 = new Menu();
                    frm1.Show();
                }
                else if (tipo == "3")
                {
                    this.Hide();
                    Menu frm2 = new Menu();
                    frm2.Show();
                }
                else
                {
                    MessageBox.Show("Error");
                }

            }
            con.Close();
        }
